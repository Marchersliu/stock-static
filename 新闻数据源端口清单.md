import requests

def fetch_wallstreetcn_lives(channel='global-channel', limit=20):
    url = 'https://api-prod.wallstreetcn.com/apiv1/content/lives'

        'channel': channel,

### 中国新闻网（替代央视）
```python
import requests, xml.etree.ElementTree as ET

def fetch_chinanews_finance(max_items=8):
    url = 'https://www.chinanews.com/rss/finance.xml'
    resp = requests.get(url, headers={'User-Agent': 'Mozilla/5.0'})
    root = ET.fromstring(resp.text)
    items = []
    for item in root.findall('.//item')[:max_items]:
        items.append({
            'title': item.findtext('title', ''),
            'date': item.findtext('pubDate', ''),
            'source': '中国新闻网',
            'sourceClass': 'chinanews',
            'category': 'policy',
            'url': item.findtext('link', ''),
        })
    return items
```
